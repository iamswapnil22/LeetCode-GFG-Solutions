```
class Solution {
public:
    int n;
    int solve(vector<int> &h,int i,int bricks,int ladders)
    {
        if(i == n-1)
            return 0;

        if(h[i+1] <= h[i])
        {
            return 1+solve(h,i+1,bricks,ladders);
        }
        else
        {
            int b = 0;
            int l = 0;

            if(bricks >= h[i+1] - h[i] )
            {
                b = 1 + solve(h,i+1,(bricks-(h[i+1]-h[i])),ladders);
            } 

            if(ladders > 0)
                l = 1 + solve(h,i+1,bricks,ladders-1);
            return max(l,b);
        }
        return 0;
    }

    int furthestBuilding(vector<int>& h, int bricks, int ladders) {
        n=h.size();
        return solve(h,0,bricks,ladders);
    }
};
```

## Accepted Solution using Lazy Greedy

```
class Solution {
public:
    int n;
    priority_queue<int> q;

    int furthestBuilding(vector<int>& h, int bricks, int ladders) {
        n=h.size();
        int i=0;
        //n-1 because we are checking h[i+1]
        for(;i<n-1;i++)
        {
            if(h[i+1] <= h[i])
            {
                continue;
                //if next building is same or small we easily move forward
            }

            int diff = h[i+1] - h[i];
            if(bricks >= diff)
            { 
                bricks-=diff;
                q.push(diff);
            }else if(ladders > 0){
                if(!q.empty())
                {
                    int past_bricks = q.top();
                    if(past_bricks > diff)
                    {
                        bricks+=past_bricks;
                        q.pop();
                        bricks-=diff;
                        q.push(diff);
                    }
                }
                ladders--;
            }else
            {
                break;
            }
                
            

        }
        return i;
    }
};
```
